Configuring MongoDB Shell (mongosh) Prompt (.mongoshrc.js)

Apr 23, 2024

We can customize mongosh prompt through the .mongoshrc.js file which needs to be present in the HOME directory. This file works in windows as well as linux.

If you’re uncertain about the location of your HOME directory, a simple command will reveal its path on both Windows and Linux environments.

echo $HOME

Chuck this in .mongoshrc.js in your $HOME directory.

(() => {
    try {
        let hostString = '';
        const uri = db.getMongo()._uri;
        // Check if uri contains credentials
        let startIndex = uri.indexOf('@');
        if (startIndex == -1) {
            startIndex = uri.indexOf('//');
            const endIndex = uri.indexOf('/', startIndex + 2);
            hostString = uri.slice(startIndex + 2, endIndex);
        } else {
            const endIndex = uri.indexOf('/', startIndex);
            hostString = uri.slice(startIndex + 1, endIndex);
        }

        prompt = function () {
            return hostString + ' : ' + db.getName() + ' > ';
        }
    } catch (error) {
        console.log('Looks like non mongosh environment or something is wrong.');
        prompt = undefined;
    }
})();

After applying the change mongosh prompt will look something like this:

127.0.0.1:27017 : test >
